Incremental Player Data Replication

local function BuildChangedData(ToSend, Old, Current)
	local HasChanges = false

	for Key, Value in Current do
		local OldValue = Old[Key]

		if typeof(Value) == "table" then
			if typeof(OldValue) == "table" then
				local ChildChanges = {}

				if BuildChangedData(ChildChanges, OldValue, Value) == true then
					ToSend[Key] = ChildChanges
					HasChanges = true
				end
			else
				ToSend[Key] = table.clone(Value)
				HasChanges = true
			end
		elseif Value ~= OldValue then
			ToSend[Key] = Value
			HasChanges = true
		end
	end

	for Key in Old do
		if Current[Key] == nil then
			ToSend[Key] = "NILL"
			HasChanges = true
		end
	end

	return HasChanges
end

local function SmoothDampVector(Current, Target, Velocity, SmoothTime, MaxSpeed, DeltaTime)
	SmoothTime = math.max(0.001, SmoothTime)

	local Omega = 2 / SmoothTime
	local Change = Current - Target
	local OriginalTarget = Target
	local MaxChange = MaxSpeed * SmoothTime

	if Change.Magnitude > MaxChange then
		Change = Change.Unit * MaxChange
	end

	Target = Current - Change

	local X = Omega * DeltaTime
	local Exp = 1 / (1 + X + (0.48 * X * X) + (0.235 * X * X * X))
	local Temp = (Velocity + (Omega * Change)) * DeltaTime
	local NewVelocity = (Velocity - (Omega * Temp)) * Exp
	local Output = Target + ((Change + Temp) * Exp)

	if (OriginalTarget - Current):Dot(Output - OriginalTarget) > 0 then
		Output = OriginalTarget
		NewVelocity = Vector3.new(0, 0, 0)
	end

	return Output, NewVelocity
end

local function GetFormationOffset(Index, Count, Forward, Right, Spacing)
	if Count == 1 then
		return -Forward * Spacing
	end

	local Radius = math.max(Spacing, Spacing / (2 * math.sin(math.pi / Count)))
	local Angle = (math.pi * 0.5) + (((Index - 1) / Count) * math.pi * 2)

	return (Right * math.cos(Angle) * Radius) + (Forward * math.sin(Angle) * Radius)
end
